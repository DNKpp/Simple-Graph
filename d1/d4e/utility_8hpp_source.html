330 template <vertex_descriptor TVertex, std::invocable<TVertex, TVertex> TWeightCalculator>
331 struct weighted_node_factory_t
332 {
333 using vertex_t = TVertex;
334 using weight_t = weight_type_of_t<TWeightCalculator, TVertex>;
335 using node_t = weighted_node<vertex_t, weight_t>;
336
337 TWeightCalculator weightCalculator{};
338
339 constexpr node_t operator ()(const node_t& predecessor, const TVertex& cur_vertex)
340 {
341 weight_t rel_weight{ std::invoke(weightCalculator, predecessor.vertex, cur_vertex) };
342 assert(weight_t{} <= rel_weight && "relative weight between nodes must be greater or equal zero.");
343
344 return
345 {
346 .predecessor = predecessor.vertex,
347 .vertex = cur_vertex,
348 .weight_sum = predecessor.weight_sum + rel_weight
349 };
350 }
351 };
