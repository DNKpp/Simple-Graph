64 template <class TNode, vertex_descriptor TVertex = node_vertex_t<TNode>, weight TWeight = node_weight_t<TNode>>
65 void dynamic_cost_traverse
66 (
67 node_factory_for<TNode> auto&& nodeFactory,
68 TNode begin,
69 neighbor_searcher_for<TVertex> auto&& neighborSearcher,
70 node_callback<TNode> auto&& callback,
71 vertex_predicate_for<TNode> auto&& vertexPredicate,
72 state_map_for<TVertex, dynamic_cost_state_t<TWeight>> auto&& stateMap,
73 open_list_for<TNode> auto&& openList
74 )
75 {
76 using state_t = dynamic_cost_state_t<TWeight>;
77
78 assert(detail::empty_helper(openList));
79
80 stateMap[begin.vertex] = { visit_state::discovered, TWeight{} };
81 emplace(openList, std::move(begin));
82
83 while (!detail::empty_helper(openList))
84 {
85 TNode predecessor{ take_next(openList) };
86 if (visit_state::visited == std::exchange(std::get<0>(stateMap[predecessor.vertex]), visit_state::visited))
87 continue;
88
89 if (detail::shall_interrupt(callback, predecessor))
90 return;
91
92 for (const TVertex& cur_vertex : std::invoke(neighborSearcher, predecessor.vertex))
93 {
94 auto&& [cur_state, cur_weight] = stateMap[cur_vertex];
95 if (cur_state == visit_state::visited || !std::invoke(vertexPredicate, predecessor, cur_vertex))
96 continue;
97
98 TNode current{ std::invoke(nodeFactory, predecessor, cur_vertex) };
99 switch (cur_state)
100 {
101 case visit_state::none:
102 cur_state = visit_state::discovered;
103 cur_weight = static_cast<TWeight>(current);
104 emplace(openList, std::move(current));
105 break;
106
107 case visit_state::discovered:
108 if (current.weight_sum < cur_weight)
109 {
110 cur_weight = static_cast<TWeight>(current);
111 emplace(openList, std::move(current));
112 }
113 break;
114 }
115 }
116 }
117 }
