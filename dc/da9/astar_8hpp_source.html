20 template <vertex_descriptor TVertex, weight TWeight>
21 struct astar_node
22 {
23 using vertex_type = TVertex;
24 using weight_type = TWeight;
25
26 std::optional<TVertex> predecessor{};
27 TVertex vertex{};
28 TWeight weight_sum{};
29 TWeight weight_estimated{};
30
31 [[nodiscard]]
32 constexpr bool operator ==(const astar_node& other) const noexcept = default;
33
34 [[nodiscard]]
35 constexpr std::compare_three_way_result_t<weight_type> operator <=>(const astar_node& other) const noexcept
36 {
37 return static_cast<TWeight>(*this) <=> static_cast<TWeight>(other);
38 }
39
40 [[nodiscard]]
41 explicit constexpr operator TWeight() const noexcept
42 {
43 return weight_sum + weight_estimated;
44 }
45 };

47 template <vertex_descriptor TVertex, std::invocable<TVertex, TVertex> TWeightCalculator, std::invocable<TVertex> THeuristic>
48 struct astar_node_factory_t
49 {
50 using vertex_t = TVertex;
51 using weight_t = weight_type_of_t<TWeightCalculator, TVertex>;
52 using node_t = astar_node<vertex_t, weight_t>;
53
54 TWeightCalculator weightCalculator{};
55 THeuristic heuristic{};
56
57 constexpr node_t operator ()(const node_t& predecessor, const TVertex& cur_vertex)
58 {
59 weight_t rel_weight{ std::invoke(weightCalculator, predecessor.vertex, cur_vertex) };
60 assert(weight_t{} <= rel_weight && "relative weight between nodes must be greater or equal zero.");
61
62 weight_t estimated_weight{ std::invoke(heuristic, cur_vertex) };
63 assert(weight_t{} <= estimated_weight && "estimated weight must be greater or equal zero.");
64
65 return
66 {
67 .predecessor = predecessor.vertex,
68 .vertex = cur_vertex,
69 .weight_sum = predecessor.weight_sum + rel_weight,
70 .weight_estimated = estimated_weight
71 };
72 }
73 };

75 template <vertex_descriptor TVertex, class TWeightCalculator, class THeuristic>
76 [[nodiscard]]
77 constexpr astar_node_factory_t<TVertex, TWeightCalculator, THeuristic> make_astar_node_factory
78 (
79 TWeightCalculator&& weightCalc,
80 THeuristic&& heuristic
81 )
82 {
83 return { .weightCalculator = std::forward<TWeightCalculator>(weightCalc), .heuristic = std::forward<THeuristic>(heuristic) };
84 }
