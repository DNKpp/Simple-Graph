40 TDestinationPredicate destinationPredicate{};
41 TOutItr outItr{};
42 TPredecessorMap predecessors{};
43 bool success{ false };
44
45 [[nodiscard]]
46 bool operator ()(const TNode& node)
47 requires requires
48 {
49 node.vertex;
50 node.predecessor;
51 { std::invoke(destinationPredicate, node) } -> std::convertible_to<bool>;
52 predecessors[node.vertex] = node.predecessor;
53 }
54 {
55 predecessors[node.vertex] = node.predecessor;
56
57 if (std::invoke(destinationPredicate, node))
58 {
59 success = true;
60 *outItr = node.vertex;
61 ++outItr;
62 for
63 (
64 auto predecessor = predecessors[node.vertex];
65 predecessor;
66 predecessor = predecessors[*predecessor], ++outItr
67 )
68 {
69 *outItr = *predecessor;
70 }
71 return true;
72 }
73 return false;
74 }
75 };

94
95 std::invoke(find_path_callback, std::ref(pathFinder));
